package ipl.win

import
   graphics(Window),
   io(stop)

procedure win(width, height)

   /width := 500
   /height := 500

   return Window().set_size(width, height).set_canvas("normal") |
      stop("*** can't open window")

   return

end

procedure winf(width, height)

   /width := 500
   /height := 500

   return Window().set_size(width, height).set_canvas("normal")

end
