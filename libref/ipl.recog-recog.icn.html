package ipl.recog

import
   io(read, write, writes)

procedure main(init, goal)
   local line

   init()
   while line := read() do {
      writes(image(line))
      if line ? (goal() & pos(0)) then 
         write(": accepted")
      else write(": rejected")
      }
end
