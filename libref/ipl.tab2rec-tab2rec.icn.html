#
#     This procedure to takes tab-separated strings and inserts them
#  into fields of a record.
#
#  Vertical tabs in strings are converted to carriage returns.
#
#  (Works for lists too.)
#
############################################################################

package ipl.tab2rec

procedure tab2rec(s, x)
   local i

   i := 0

   s ? {
      while x[i +:= 1] := map(tab(upto('\t') | 0), "\v", "\n") do
         move(1) | break
      }

   return

end
