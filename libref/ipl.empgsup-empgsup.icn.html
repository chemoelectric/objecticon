package ipl.empgsup

procedure _Initialize(limit)
   local itime, t1, t3
   
   itime := &time

   every 1 to limit do {
      &null
      }

   t1 := (&time - itime)

   itime := &time

   every 1 to limit do {
      &null & &null
      }

   t3 := (&time - itime)

   return (t1 + t3) / 2

end
