#
#  This procedure writes fields of a record as tab-separated string.
#  Carriage returns in files are converted to vertical tabs.
#  (Works for lists too.)
#
############################################################################

package ipl.rec2tab

import
   io(write, writes)

procedure rec2tab(x)
   local i

   i := *x - 1
   every writes(map(\!x, "\n", "\v"),"\t") \ i
   write(map(\x[-1], "\n", "\v")) | write()

   return

end
