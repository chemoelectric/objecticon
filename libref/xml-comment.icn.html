package xml

#
#  This class represents a comment.
#
class Comment(Node)
   public 
      comment

   #
   # Get the comment body.
   #
   public get_comment()
      return comment
   end

   #
   # Set the comment body
   #
   public set_comment(comment)
      self.comment := comment
      link
   end

   public override new(s)
      Node.new()
      comment := s
      return
   end
end
