#  Icon Analyst 35.

package ipl.verncnt

# number of n-digit versum numbers
procedure verncnt(n)		

   return case integer(n) of {
      1       :        4
      2       :       14
      3       :       93
      4       :      256
      5       :     1793
      6       :     4872
      7       :    34107
      8       :    92590
      9       :   648154
      10      :  1759313
      default :  19 * verncnt(n - 2)
      }

end
