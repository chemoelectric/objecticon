package ipl.colrspec

import
   graphics(Window)

procedure colrspec(s)
   local color
   static max

   initial {
      max := real(2 ^ 16 - 1)
      }

   color := ""

   Window.color_value(s) ? {
      every 1 to 3 do {
         color ||:= (tab(upto(',') | 0) / max) || " "
         move(1)
         }
      return color
      }

   fail

end
