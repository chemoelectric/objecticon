#  distances d1, d2, d3, ... from the origin.
#
############################################################################

package ipl.distance

import 
   util(Math)

procedure distance(d[])
   local sum

   sum := 0

   every sum +:= !d ^ 2

   return Math.sqrt(sum)

end
