#  graphics, or (b) if it is, the graphics system is not running.
#
############################################################################

package ipl.ifg

import
   graphics(Window),
   io(close)

procedure ifg()
   local win

   if (&features == "graphics") &
      win := Window() then {
         close(win)
         return
         }
      
   else fail

end
