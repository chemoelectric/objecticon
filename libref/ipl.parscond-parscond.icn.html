#
#     The necessary record declaration is provided by the program with which
#  is linked.
#
############################################################################

#  See also:  parsgen.icn
#

package ipl.parscond

procedure parscond(R)
   local result, x

   result := ""

   every x := !(R.alts) do
     result ||:= string(x) | parscond(x)
   
   return result

end
