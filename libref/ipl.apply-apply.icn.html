#
#  This procedure applies a list of functions to an argument.  An example is
#
#	apply([integer, log], 10)
#
#  which is equivalent to integer(log(10)).
#
#
############################################################################

package ipl.apply

procedure apply(plist, arg)
   local p

   plist := copy(plist)

   p := get(plist) | fail

   if *plist = 0 then
      suspend p(arg)
   else
      suspend p(apply(plist, arg))

end
