package xml

#
# This class represents a CDATA section.
#
class CData(Node)
   public content

   #
   # Get the content
   #
   public get_content()
      return content
   end

   #
   # Set the content
   #
   public set_content(content)
      self.content := content
      link
   end

   public override new(s)
      Node.new()
      content := s
      return
   end
end
