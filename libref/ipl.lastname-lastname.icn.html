#  Produces the last name of a name in conventional form.  Obviously, it
#  doesn't work for every possibility.
#
############################################################################

package ipl.lastname

procedure lastname(s)
   local line, i

   line := trim(s)
   line ?:= tab(upto(','))		# Get rid of things like " ... , Jr."
   line ? {
      every i := upto(' ')
      tab(\i + 1)
      return tab(0)
      }

end
