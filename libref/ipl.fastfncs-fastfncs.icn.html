package ipl.fastfncs

import
   ipl.factors(factorial),
   util(Math)

procedure g(k, n)
   local value
   static psi

   initial psi := 1.0 / Math.PHI

   if n = 0 then return 0

   value := 0

   value +:= integer(psi * integer((seq(0) \ k + n) / real(k)) + psi)

   return value

end 

procedure robbins(n)
   local numer, denom, i

   numer := denom := 1

   every i := 0 to n - 1 do {
      numer *:= factorial(3 * i + 1)
      denom *:= factorial(n + i)
      }

   return numer / denom

end
