############################################################################

package ipl.pascal

import
   io(write, writes),
   ipl.math(binocoef)

#  The Pascal Triangle

# Pascal triangle
procedure pascal(n)		
   local i, j

   /n := 16

   write("width=", n, " height=", n)	# carpet header

   every i := 0 to n - 1 do {
      every j := 0 to n - 1 do
         writes(binocoef(i, j) | 0, " ")
      write()
      }

end
