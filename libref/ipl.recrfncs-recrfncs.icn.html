package ipl.recrfncs

procedure acker(i, j)

   if i = 0 then return j + 1
   if j = 0 then  return acker(i - 1, 1)
   else return acker(i - 1, acker(i, j - 1))

end

procedure fib(i)

   if i = (1 | 2) then return 1

   else return fib(i - 1) + fib(i - 2)

end

procedure q(i)

   if i = (1 | 2) then return 1
   else return q(i - q(i - 1)) + q(i - q(i - 2))

end
