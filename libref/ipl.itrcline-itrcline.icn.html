package ipl.itrcline

import
   io(read)

# generate trace messages in file
procedure itrcline(f)		
   local line

   while line := read(f) do
      line ? {
         if (="             :" & move(6) & ="main") | (move(12) & ":      |")
         then suspend line
         }

end
