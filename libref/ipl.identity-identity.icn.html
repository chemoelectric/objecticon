package ipl.identity

procedure identity(x)

   return case x of {
      "null":  &null
      "integer":  0
      "real":  0.0
      "string":  ""
      "ucs": u""
      "cset":  ''
      "list":  []
      "set":  set()
      "table":  table()
      default: fail
      }

end
