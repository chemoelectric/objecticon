package xml

#
# This represents the DocumentType declaration.
#
class DocType(Node)
   public
      name,
      external_id

   #
   # Get the name
   #
   public get_name()
      return name
   end

   #
   # Set the name
   #
   public set_name(s)
      name := s
      link
   end

   #
   # Get the ExternalID declared, if any
   #
   public get_external_id()
      return external_id
   end

   #
   # Set the external ID as an ExternalID object.
   #
   public set_external_id(x)
      external_id := x
      link
   end

   public override new(n, i)
      Node.new()
      self.name := n
      self.external_id := i
      return
   end
end
