package ipl.divide

# long division
procedure divide(i, j, k)		
   local q, pad

   /k := 5

   q := ""

   pad := 20

   i ||:= repl("0", pad)

   every 1 to k do {
      q ||:= i / j
      i %:= j
      if i = 0 then break
      }

   return q[1:-pad]

end
