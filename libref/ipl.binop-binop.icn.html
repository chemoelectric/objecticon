#  This procedure applies a binary operation to a list of arguments.
#  For example,
#
#	binop("+", 1, 2, 3)
#
#  returns 6.
#
############################################################################

package ipl.binop

# apply binary operation
procedure binop(op, result, rest[])		

   every result := op(result, !rest)

   return result

end
