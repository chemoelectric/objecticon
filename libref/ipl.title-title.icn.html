#
#    The process is imperfect.
#
############################################################################
#
#  Links:  titleset
#
############################################################################

package ipl.title

import
   ipl.titleset(titleset)

procedure title(name)
   local result
   static titles

   initial titles := titleset()

   result := ""

   name ? {
      while result ||:= =!titles || " " do
        tab(many(' \t'))
      return result ? tab(-1 | 0)
      }

end
